<body>

    <header>

        <div style="font-weight: bold; color: var(--prim);">PRIVATE AI HUB</div>

        <div id="status">Initializing...</div>

        <div id="progress-container"><div id="progress-bar"></div></div>

    </header>

 

    <div id="chat-history"></div>

 

    <div class="input-area">

        <input type="text" id="user-input" placeholder="Ask Qwen something...">

        <button id="send-btn" style="background: var(--prim); border:none; padding: 10px 20px; font-weight:bold; cursor:pointer;">SEND</button>

    </div>

 

    <script type="module">

        import { pipeline, env } from 'https://cdn.jsdelivr.net/npm/@xenova/transformers@2.17.2';

 

        // --- GITHUB PAGES & SECURITY FIXES ---

        // allowLocalModels must be false to fix "Unauthorized Access" errors on GitHub Pages

        env.allowLocalModels = false;

        env.allowRemoteModels = true;

        env.useBrowserCache = true;

 

        const status = document.getElementById('status');

        const progCont = document.getElementById('progress-container');

        const progBar = document.getElementById('progress-bar');

        const history = document.getElementById('chat-history');

        const input = document.getElementById('user-input');

 

        const db = new Dexie('ChatDB');

        db.version(1).stores({ history: '++id, role, content' });

 

        let engine = null;

 

        async function init() {

            progCont.style.display = 'block';

            status.innerText = "Downloading Qwen 1.5 (WASM Mode)...";

           

            try {

                // Optimized for 3GB RAM and Legacy GPU (GeForce 210)

                engine = await pipeline('text-generation', 'Xenova/Qwen1.5-0.5B-Chat', {

                    device: 'wasm', // Bypasses legacy GPU driver errors

                    dtype: 'q4',   // Critical for low-RAM stability

                    progress_callback: (p) => {

                        if (p.status === 'progress') {

                            progBar.style.width = p.progress + '%';

                            status.innerText = `Loading: ${Math.round(p.progress)}%`;

                        }

                    }

                });

                status.innerText = "System Ready (CPU Optimized)";

                progCont.style.display = 'none';

               

                const saved = await db.history.toArray();

                saved.forEach(m => render(m.role, m.content));

            } catch (e) {

                status.innerHTML = `<span style="color:red">Error:</span> Clear browser cache and refresh.`;

                console.error(e);

            }

        }

 

        async function send() {

            const text = input.value.trim();

            if (!text || !engine) return;

            input.value = '';

           

            render('user', text);

            await db.history.add({ role: 'user', content: text });

 

            const aiBox = render('ai', '...');

           

            // Qwen ChatML Format

            const prompt = `<|im_start|>user\n${text}<|im_end|>\n<|im_start|>assistant\n`;

           

            const out = await engine(prompt, {

                max_new_tokens: 100,

                temperature: 0.7,

                stop: ['<|im_end|>']

            });

           

            const response = out[0].generated_text.split('assistant\n')[1].replace('<|im_end|>', '').trim();

            aiBox.innerText = response;

            await db.history.add({ role: 'ai', content: response });

        }

 

        function render(role, content) {

            const d = document.createElement('div');

            d.className = `msg ${role}`;

            d.innerText = content;

            history.appendChild(d);

            history.scrollTop = history.scrollHeight;

            return d;

        }

 

        document.getElementById('send-btn').onclick = send;

        input.onkeydown = (e) => { if(e.key==='Enter') send(); };

        init();

    </script>
